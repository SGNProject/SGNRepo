1function C_dx  = C_dx_func(y,p_n_i)
 2
 3    [row,col] = size(p_n_i);

 5
 6    % taking last transponder position

 8    
 9    % col is the number of transponders
10    
11    % taking last pseudorange measurement

13

18    for i = 1:1:(col-1)

20        p_(i,:) = (p_m - p_n_i(:,i))';
21        
22        y_(i,1) = y(i,1) - y_m;
23        
24    end

27
28end
