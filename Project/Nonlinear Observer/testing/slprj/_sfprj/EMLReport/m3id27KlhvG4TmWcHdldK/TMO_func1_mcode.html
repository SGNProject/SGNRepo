1function [beta_dot_stim,p_n_delta_dot_stim,v_n_dot_stim,f_n_stim,csi_dot] =  TMO_func(delta,C_dx,K,J_stim,R_nb_stim, v_n_stim,f_b_imu, beta_stim, p_n_delta_stim,sigma,K_p,csi)
 2
 3 K_pp = K(1:3,:); 
 4 K_beta_p = K(4,:);
 5 K_v_p = K(5:7,:);
 6 K_eps_p = K(8:10,:);
 7 %obtained by Riccati equation..
 8
 9
10% remember to insert a clock to define dimensions of all variables
11
12g_n = [0 0 1]';
13
14delta_diff = zeros(3,1);
15
16x_stim = [p_n_delta_stim, beta_stim];
17delta_stim = 2*C_dx*x_stim;
18delta_diff = (delta - delta_stim);
19
20
21
22p_n_delta_dot_stim = v_n_stim + K_pp * delta_diff;        % eq 14
23
24
25beta_dot_stim = K_beta_p * delta_diff;                       % eq 15
26
27
28%theoretically we can't know epsilon
29f_n_stim = R_nb_stim*f_b_imu + csi ;         % eq 18
30
31v_n_dot_stim  = f_n_stim + g_n + K_v_p*delta_diff;         % eq 16
32
33csi_dot   = -sigma*K_p*J_stim*f_b_imu + K_eps_p*delta_diff; % eq 17
34end
